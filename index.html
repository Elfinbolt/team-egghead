                        void printArray_char(char a[], int size) {
                            printf("( ");
                            for (int i = 0; i < size; i++) {
                                printf("%c", a[i]);
                                if (i < size - 1) printf(", ");
                            }
                            printf(" )\n");
                        }

                        void visualizeStep(char a[], int low, int high) {
                            printf("( ");
                            for (int i = low; i <= high; i++) {
                                printf("%c", a[i]);
                                if (i < high) printf(", ");
                            }
                            printf(" )\n");
                        }

                        void merge_char(char a[], char t[], int low1, int high1, int low2, int high2) {
                            int i = low1, j = low2, k = 0;
                            while (i <= high1 && j <= high2) {
                                if (a[i] <= a[j]) t[k++] = a[i++];
                                else t[k++] = a[j++];
                            }
                            while (i <= high1) t[k++] = a[i++];
                            while (j <= high2) t[k++] = a[j++];
                        }
                        
                        void mergeSort_char(char a[], int low, int high) {
                            if (low < high) {
                                int mid = low + (high - low) / 2;
                                char *temp = (char *)malloc((high - low + 1) * sizeof(char));
                                visualizeStep(a, low, high);
                                mergeSort_char(a, low, mid);
                                mergeSort_char(a, mid + 1, high);
                                merge_char(a, temp, low, mid, mid + 1, high);
                                for (int i = low, k = 0; i <= high; i++) a[i] = temp[k++];
                                free(temp);
                            }
                        }
                        
                        void merge_int(int a[], int t[], int low1, int high1, int low2, int high2) {

                    import pandas as pd
import matplotlib.pyplot as plt

# Read dataset
data = pd.read_csv('Viral_Social_Media_Trends.csv')

# Print column names to debug
print("Columns in CSV:", data.columns.tolist())

# Strip spaces from column names
data.columns = data.columns.str.strip()

# Check actual column names
expected_columns = ['platform', 'engagement_level']
for col in expected_columns:
    if col not in data.columns:
        print(f"⚠️ Column '{col}' not found. Check the CSV headers!")

# If 'platform' isn't found, look for similar column names
print("Dataset Preview:\n", data.head())

# Ensure correct column names
platform_col = 'platform' if 'platform' in data.columns else input("Enter correct column name for Platform: ")
engagement_col = 'engagement_level' if 'engagement_level' in data.columns else input(
    "Enter correct column name for Engagement Level: ")

# Group data
total_posts = data.groupby(platform_col).size()
high_posts = data[data[engagement_col] == 'High'].groupby(platform_col).size()
high_pct = (high_posts / total_posts * 100).sort_values(ascending=False)

# Display results
print("\nPercentage of High-Engagement Posts Per Platform:\n", high_pct)

# Generate Bar Chart
engagement_counts = data.groupby([platform_col, engagement_col]).size().unstack(fill_value=0)
engagement_pct = engagement_counts.div(engagement_counts.sum(axis=1), axis=0) * 100
engagement_pct.plot(kind='bar', figsize=(10, 6))

plt.title('Engagement Level Distribution per Platform')
plt.xlabel('Platform')
plt.ylabel('Percentage (%)')
plt.legend(title='Engagement Level')
plt.show()
